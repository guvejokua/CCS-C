#include <16F628A.h>

#FUSES NOWDT                    //No Watch Dog Timer
#FUSES NOBROWNOUT               //No brownout reset
#FUSES NOLVP                    //No low voltage prgming, B3(PIC16) or B5(PIC18) used for I/O

#use delay(crystal=16MHz)

const int ledler[] = {pin_b1, pin_b3, pin_b5, pin_b7};

void main()
{
   //Yerel Local
   //Kurulum Kýsmý
   set_tris_b(0x00);
   output_b(0x00);
   int i = 0;
   
   while(1)
   {
      //for(baþlanýç; koþul; artým))
   for(i = 0; i < 4; i++) {
      output_high(ledler[i]);
   }
   delay_ms(10);
  for(i = 0; i < 4; i++) {
      output_low(ledler[i]);
   }
   delay_ms(10);
   }
}
