#include <16F628A.h>

#FUSES NOWDT                    //No Watch Dog Timer
#FUSES NOBROWNOUT               //No brownout reset
#FUSES NOLVP                    //No low voltage prgming, B3(PIC16) or B5(PIC18) used for I/O

#use delay(crystal=16MHz)
#define led3 pin_b5
//GLobal
const int led4 = pin_b7;

void main()
{
   //Yerel Local
   //Kurulum Kýsmý
   set_tris_b(0x00);
   output_b(0x00);
   //set_tris_b(0x00);
   int led1 = pin_b1;
   int led2 = pin_b3;
   
   while(1)
   {
      output_high(led1);
      output_high(led2);
      output_high(led3);
      output_high(led4);
      delay_ms(10);
      output_low(led1);
      output_low(led2);
      output_low(led3);
      output_low(led4);
      delay_ms(10);
   }
}
